  <div class="container">
    <h1>API Object Viewer</h1>

    <label>Select Item by ID:</label>
    <select id="itemSelect">
      <option value="">-- Choose ID --</option>
    </select>

    <h3 style="margin-top:20px;">Details:</h3>
    <pre id="output">Select an ID to view details...</pre>
  </div>

  <script>
    const apiUrl = "https://api.restful-api.dev/objects";
    const selectBox = document.getElementById("itemSelect");
    const output = document.getElementById("output");
    let items = [];

    // Fetch API data
    fetch(apiUrl)
      .then(res => res.json())
      .then(data => {
        items = data;
        data.forEach(item => {
          const opt = document.createElement("option");
          opt.value = item.id;
          opt.textContent = item.id;
          selectBox.appendChild(opt);
        });
      });

    // Display selected item
    selectBox.addEventListener("change", () => {
      const id = selectBox.value;
      const found = items.find(item => item.id === id);

      output.textContent = found ? JSON.stringify(found, null, 2) : "";
    });
  </script>
